Usage

Globals:

Just follow the example and use it!


HTML:

<html>
	<head>
		<script src="https://cdn.jsdelivr.net/npm/vue@2.4.4/dist/vue.min.js"></script>
		<script src="https://cdn.jsdelivr.net/npm/framevuerk@latest/dist/framevuerk-en.min.js"></script>
		<link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/framevuerk@latest/dist/framevuerk-en.min.css"/>
		<script>
			Vue.use(framevuerk);
			new Vue({
				el: "#app"
			});
		</script>
	</head>
	<body>
		<fv-main id="app">
			<!-- Your Content -->
		</fv-main>
	</body>
</html>

CommonJS:


JS:

var Vue = require('vue');
var framevuerk = require('framevuerk/dist/framevuerk-en.js');

Vue.use(framevuerk);
new Vue({
	el: "#app"
});

Template:

<html>
	<head>
		<script src="path/to/your/app.js"></script>
		<link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/framevuerk@latest/dist/framevuerk-en.css"/>
	</head>
	<body>
		<fv-main id="app">
			<!-- Your Content -->
		</fv-main>
	</body>
</html>

ES6 (with webpack):


JS:

import Vue from 'vue'
import framevuerk from 'framevuerk/dist/framevuerk-en.js'
import 'framevuerk/dist/framevuerk-en.css'

Vue.use(framevuerk);
new Vue({
	el: "#app"
});

Template via pug (after including app.js):

fv-main#app
	 // Your Content
